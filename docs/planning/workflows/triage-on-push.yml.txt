name: Triage On Push (Template)

on:
  push:
    branches: [ main ]
  pull_request:

jobs:
  lightweight-contracts:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-python@v5
        with:
          python-version: '3.10'
      - run: python -m pip install --upgrade pip jsonschema
      - name: Run viewer contract (no DDS)
        run: |
          python run_scripts/test_viewer_contract.py

  remote-dds-triage:
    needs: [lightweight-contracts]
    runs-on: ubuntu-latest
    if: always()  # still try triage even if lightweight job fails, to collect signal
    steps:
      - name: Run triage suite on DDS host via SSH
        uses: appleboy/ssh-action@v1.2.0
        with:
          host: ${{ secrets.DDS_SSH_HOST }}
          username: ${{ secrets.DDS_SSH_USER }}
          key: ${{ secrets.DDS_SSH_KEY }}
          script_stop: true
          script: |
            set -e
            cd ${{ secrets.DDS_WORKDIR }}
            export OPENAI_API_KEY=${{ secrets.OPENAI_API_KEY }}
            export ANTHROPIC_API_KEY=${{ secrets.ANTHROPIC_API_KEY }}
            bash run_scripts/run_triage_suite.sh || echo TRIAGE_FAILED=1 > triage_status.env
            tail -n +1 logs/*.log | sed -e 's/^/REMOTE_LOG: /' | tail -n 2000 || true
      - name: Fetch logs from DDS host
        if: always()
        uses: appleboy/scp-action@v0.1.7
        with:
          host: ${{ secrets.DDS_SSH_HOST }}
          username: ${{ secrets.DDS_SSH_USER }}
          key: ${{ secrets.DDS_SSH_KEY }}
          source: ${{ secrets.DDS_WORKDIR }}/logs/*.log
          target: logs/
          strip_components: 2
      - name: Upload logs as artifact
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: dds-triage-logs
          path: logs/*.log
      - name: Comment on PR on failure
        if: failure()
        uses: actions/github-script@v7
        with:
          script: |
            const body = `DDS triage failed. Logs attached as artifacts.`;
            if (context.payload.pull_request) {
              await github.rest.issues.createComment({
                owner: context.repo.owner,
                repo: context.repo.repo,
                issue_number: context.payload.pull_request.number,
                body,
              });
            } else {
              await github.rest.repos.createCommitComment({
                owner: context.repo.owner,
                repo: context.repo.repo,
                commit_sha: context.sha,
                body,
              });
            }
