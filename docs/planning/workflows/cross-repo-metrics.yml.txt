name: Cross-Repo Metrics (Template)

on:
  schedule:
    - cron: '0 6 * * *'  # daily 06:00 UTC
  workflow_dispatch:

jobs:
  gather:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Checkout genesis-lib
        uses: actions/checkout@v4
        with:
          repository: org/genesis-lib
          path: external/genesis-lib
          token: ${{ secrets.CROSS_REPO_TOKEN }}
      - name: Checkout genesis-examples
        uses: actions/checkout@v4
        with:
          repository: org/genesis-examples
          path: external/genesis-examples
          token: ${{ secrets.CROSS_REPO_TOKEN }}
      - uses: actions/setup-python@v5
        with:
          python-version: '3.10'
      - run: python -m pip install --upgrade pip
      - name: Compute metrics (lib)
        run: |
          python docs/tools/generate_repo_metrics.py external/genesis-lib > metrics/genesis-lib.json
      - name: Compute metrics (examples)
        run: |
          python docs/tools/generate_repo_metrics.py external/genesis-examples > metrics/genesis-examples.json
      - name: Commit results
        run: |
          git config user.name "metrics-bot"
          git config user.email "metrics-bot@users.noreply.github.com"
          git add metrics/
          git commit -m "chore(metrics): update cross-repo metrics" || echo "No changes"
          git push
